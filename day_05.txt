-- lon nhat -> select max(...)
-- nho nhat -> select min(...)

-- VIET CAU LENH SQL TIM GIA THAP NHAT CUA CAC SAN PHAM TRONG BANG PRODUCTS
SELECT  MIN([UnitPrice]) AS "GIA THAP NHAT"
FROM [dbo].[Products];

-- VIE CAU LENH LAY RA NGAY DAT HANG GAN DAY NHAT TU BANG ORDERS
SELECT MAX([OrderDate]) AS "NGAY DAT HANG GAN DAY NHAT"
FROM [dbo].[Orders];

-- VIET CAU LENH SQL TIM 
--CO SO LUONG HANG TON KHO (UNITSLNSTOCK) LON NHAT
SELECT MAX([UnitsInStock]) AS [HANG TON KHO NHIEU NHAT LA]
FROM [dbo].[Products];

-- HAY CHO BIET TUOIO DOI CUA NHAN VIEN LON NHAT CONG TY 
-- GOI Y AI CO NGAY SINH CANG NHO THI NGUOI DO CANG LON TUOI 
SELECT MIN([BirthDate]) AS [LON TUOI NHAT]
FROM [dbo].[Employees];

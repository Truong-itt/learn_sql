-- LAY RA SO LUONG KHACH HANG CO TRONG DU LIEU 
SELECT COUNT(*) AS [SO LUONG KHACH HANG]
FROM [dbo].[Customers];

-- TINH TONG SO TIEN VAN CHUEN CUA TAT CA CAC DON DAT HANG
SELECT SUM([Freight]) AS [TOGN SO TIENF VAN CHUYEN CUA ON HANG]
FROM [dbo].[Orders];

--TINH TRUNG BINFH SO LUONG DAT HANG CUA TAT CA CAC SAN PHAM TRONG BANG
SELECT AVG([Quantity]) AS [TRUNG BINH SO LUONG DAT HANG]
FROM [dbo].[Order Details];

--DEM SO LUONG ,TINH TONG SO LUONG HANG TRONG KHO VA TRUNG BINFH GIA CUA 
-- SAN PHAM CO TRONG BANG PRODUCT
SELECT COUNT(*),SUM([UnitsInStock]),AVG([UnitPrice])
FROM [dbo].[Products];
--DEM SO LUONG DON HANG TU BANG ORDER VOI 2 CHA DUNG * VA DUNG MA DON HANG
SELECT COUNT(*),COUNT([CustomerID])
FROM[dbo].[Orders];

-- TU BANG ORDER DEITAILS HAY TINH TRUNG BINH CHO COT UNIPRICE
-- VA TINH TONG CHO COT QUANTITY
SELECT AVG([UnitPrice]),SUM([Quantity])
FROM [dbo].[Order Details];
